    <script>
        // Global Constants
        const ARDUINO_URL = 'http://192.168.1.100/activate_drone'; // Placeholder URL
        const SCREEN_IDS = ['screen-intro', 'screen-dashboard', 'screen-map', 'screen-drone-status'];

        // DOM Elements
        const notifier = document.getElementById('notifier');
        const instructionText = document.getElementById('current-instruction');
        const canvas = document.getElementById('routeCanvas');
        const ctx = canvas.getContext('2d');
        const activationModal = document.getElementById('activation-options-modal');

        // TTS setup
        const speech = window.speechSynthesis;
        let voice;
        // Function to set voice once loaded
        const setVoice = () => {
            voice = speech.getVoices().find(v => v.lang.startsWith('en')) || speech.getVoices()[0];
        };

        // State Variables
        let currentHazard = 'none'; // 'none', 'fire', or 'flood'
        // Map data: Coordinates (x, y) based on a 800x600 internal canvas scale
        const mapScaleX = 800;
        const mapScaleY = 600;
        const startPoint = { x: 80, y: 450 }; // User location
        const exitPoint = { x: 750, y: 150 }; // Nearest exit

        // Base route without hazards (Path A: Straight line, Path B: Longer route)
        const pathA = [startPoint, exitPoint];
        const pathB = [startPoint, { x: 150, y: 50 }, { x: 700, y: 50 }, exitPoint];

        // --- Core Functions ---

        /**
         * Converts base coordinates to canvas coordinates (responsive).
         * @param {number} base_x - X coordinate from 800-width map.
         * @param {number} base_y - Y coordinate from 600-height map.
         * @returns {{x: number, y: number}} Scaled coordinates.
         */
        function scaleCoords(base_x, base_y) {
            const container = canvas.parentElement;
            const width = container.offsetWidth;
            const height = container.offsetHeight;

            return {
                x: (base_x / mapScaleX) * width,
                y: (base_y / mapScaleY) * height
            };
        }

        /**
         * Updates canvas dimensions and redraws content on resize.
         */
        function resizeCanvas() {
            const container = canvas.parentElement;
            canvas.width = container.offsetWidth;
            canvas.height = container.offsetHeight;

            // Only redraw if the map screen is visible
            if (document.getElementById('screen-map').classList.contains('hidden') === false) {
                drawMap();
            }
            // Move user marker to scaled position
            const scaledStart = scaleCoords(startPoint.x, startPoint.y);
            const marker = document.getElementById('user-location-marker');
            marker.style.left = `${scaledStart.x - (marker.offsetWidth / 2)}px`;
            marker.style.top = `${scaledStart.y - marker.offsetHeight}px`;
        }
        window.addEventListener('resize', resizeCanvas);


        /**
         * Handles screen navigation.
         * @param {string} screenId - The data-screen identifier.
         */
        function navigateTo(screenId) {
            SCREEN_IDS.forEach(id => {
                const screen = document.getElementById(id);
                if (screen) {
                    screen.classList.add('hidden');
                    screen.classList.remove('flex');
                }
            });

            const nextScreen = document.getElementById(`screen-${screenId}`);
            if (nextScreen) {
                nextScreen.classList.remove('hidden');
                nextScreen.classList.add('flex');
            }

            if (screenId === 'map') {
                resizeCanvas(); // Ensure canvas is sized correctly
                // Initial route calculation
                recalculateRoute(currentHazard);
            }
            if (screenId === 'drone-status') {
                speak("Drone scanning hazard area. Guidance activated.");
            }
        }


        /**
         * Displays a temporary alert notification.
         * @param {string} message - The message to display.
         * @param {string} [type='orange'] - Color class ('orange' or 'red').
         */
        function showNotifier(message, type = 'orange') {
            notifier.textContent = message;
            notifier.classList.remove('bg-[#E63946]', 'bg-[#FFA500]');
            notifier.classList.add(type === 'red' ? 'bg-[#E63946]' : 'bg-[#FFA500]');

            notifier.classList.remove('-translate-y-full');
            setTimeout(() => {
                notifier.classList.add('-translate-y-full');
            }, 5000);
        }

        /**
         * Uses the Web Speech API to speak a message.
         * @param {string} text - The text to speak.
         */
        function speak(text) {
            if (!voice) {
                setVoice();
            }
            if (speech.speaking) {
                speech.cancel();
            }
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.voice = voice;
            utterance.pitch = 1;
            utterance.rate = 1.1;
            speech.speak(utterance);

            // Update UI instruction
            instructionText.textContent = text;
        }


        // --- Screen 2 Logic ---

        /**
         * Shows the modal for choosing activation mode.
         */
        function showActivationOptions() {
            activationModal.classList.remove('hidden');
        }

        /**
         * Closes the modal for choosing activation mode.
         */
        function closeActivationOptions() {
            activationModal.classList.add('hidden');
        }
        
        /**
         * Handles the user's choice of activation mode.
         * @param {string} mode - 'real' or 'simulation'.
         */
        function handleActivation(mode) {
            closeActivationOptions();
            startEmergencyFlow(mode);
        }

        /**
         * Simulates sending a command to the Arduino and initiates the emergency flow.
         * @param {string} mode - 'real' or 'simulation'.
         */
        async function startEmergencyFlow(mode) {
            // Disable button during activation
            const button = document.getElementById('activate-button');
            button.disabled = true;
            button.textContent = 'ACTIVATING...';
            button.classList.remove('animate-pulse');

            let introMsg;
            let activationText;

            if (mode === 'real') {
                introMsg = 'REAL EMERGENCY. Activating full evacuation protocol. Move to the nearest exit immediately.';
                activationText = 'Full System Activated. Evacuation route calculated.';
            } else { // simulation
                introMsg = 'Emergency simulation initiated. Activating evacuation protocol. Evacuate immediately.';
                activationText = 'Drone Activated (Simulation). Evacuation route calculated.';
            }


            // 1. Send simulated HTTP GET to Arduino
            try {
                console.log(`Attempting to send GET request to Arduino at: ${ARDUINO_URL}`);
                // const response = await fetch(ARDUINO_URL); // Commented out to prevent CORS errors in environment

                // Simulate success after a delay
                await new Promise(resolve => setTimeout(resolve, 1000));

                showNotifier(activationText, mode === 'real' ? 'red' : 'orange');
                console.log('System activation simulated successfully in mode: ' + mode);

            } catch (error) {
                // Handle expected network/CORS error for local hardware
                showNotifier('System Activation Attempted. Proceeding with guidance.', 'red');
                console.warn('Network error (expected in simulation). Proceeding...');
            }

            // 2. TTS Announcement
            speak(introMsg);
            
            // NEW: Announce drone ETA after the initial alert
            setTimeout(() => {
                speak("Drone deployed. Estimated time of arrival to your location is 1 minute.");
            }, 3000); // Wait 3 seconds to avoid interrupting the main alert

            // 3. Navigate flow: Dashboard -> Drone Status (briefly) -> Map
            setTimeout(() => navigateTo('drone-status'), 1500);
            setTimeout(() => {
                navigateTo('map'); // Return to map for immediate guidance
                button.disabled = false;
                button.textContent = 'Activate Emergency';
            }, 5000);
        }

        // --- Screen 3 Logic ---

        /**
         * Clears the canvas and draws the current evacuation route and hazards.
         */
        function drawMap() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            let route;
            let instructions;
            let hazardRects = [];

            // Determine the route based on the current hazard state
            switch (currentHazard) {
                case 'fire':
                    // Path A blocked near the exit, switch to Path B
                    route = pathB;
                    instructions = [
                        "ALERT! Fire detected at main exit. Route diverted! Turn immediately North towards the rear staircase. Remain calm.",
                        "Proceed North past the lobby, then turn East into the administrative wing.",
                        "Follow the long path to Alternate Exit Two."
                    ];
                    // Simulate fire hazard blocking exit (Top-right corner)
                    hazardRects.push({ x: 700, y: 100, w: 100, h: 100, label: 'FIRE ZONE' });
                    break;
                case 'flood':
                    // Path B blocked mid-way, switch to Path A
                    route = pathA;
                    instructions = [
                        "WARNING! Flood detected in the North wing. Route is safe. Turn left immediately toward the central corridor.",
                        "Proceed straight ahead on the central corridor.",
                        "The main exit is 20 meters ahead."
                    ];
                    // Simulate flood hazard blocking mid-path (near 700, 50)
                    hazardRects.push({ x: 600, y: 0, w: 150, h: 100, label: 'FLOOD' });
                    break;
                case 'clear':
                case 'none':
                default:
                    route = pathA;
                    instructions = [
                        // UPDATED: Clear, reassuring first instruction with direction
                        "Don't worry, the path is safe. We have confirmed the optimal route to the main exit. Please proceed straight down the current corridor.",
                        "Continue straight for 30 meters. Do not take any turns.",
                        "The main exit is now directly ahead. Move quickly and calmly."
                    ];
                    break;
            }

            // Draw Route (Green Line)
            ctx.beginPath();
            ctx.lineWidth = 10;
            ctx.strokeStyle = '#2EC4B6'; // Green
            ctx.lineJoin = 'round';
            ctx.lineCap = 'round';
            ctx.shadowColor = '#2EC4B6';
            ctx.shadowBlur = 10; // Glowing effect

            route.forEach((point, index) => {
                const scaledPoint = scaleCoords(point.x, point.y);
                if (index === 0) {
                    ctx.moveTo(scaledPoint.x, scaledPoint.y);
                } else {
                    ctx.lineTo(scaledPoint.x, scaledPoint.y);
                }
            });
            ctx.stroke();
            ctx.shadowBlur = 0; // Reset shadow for other elements

            // Draw Route Arrows (simplified: one arrow per segment)
            for (let i = 0; i < route.length - 1; i++) {
                const p1 = scaleCoords(route[i].x, route[i].y);
                const p2 = scaleCoords(route[i + 1].x, route[i + 1].y);
                drawArrow(p1, p2, 16, '#2EC4B6'); // Draw a larger arrow on the path segment
            }

            // Draw Hazards (Semi-transparent Red)
            hazardRects.forEach(hazard => {
                const scaledStart = scaleCoords(hazard.x, hazard.y);
                const scaledEnd = scaleCoords(hazard.x + hazard.w, hazard.y + hazard.h);
                const scaledW = scaledEnd.x - scaledStart.x;
                const scaledH = scaledEnd.y - scaledStart.y;

                ctx.fillStyle = 'rgba(230, 57, 70, 0.7)'; // Semi-transparent Red (#E63946)
                ctx.fillRect(scaledStart.x, scaledStart.y, scaledW, scaledH);

                // Add flashing hazard label
                ctx.fillStyle = (Math.floor(Date.now() / 500) % 2 === 0) ? '#FFFFFF' : '#FFD700'; // Flashing text
                ctx.font = 'bolder 18px Inter';
                ctx.textAlign = 'center';
                ctx.fillText(hazard.label, scaledStart.x + scaledW / 2, scaledStart.y + scaledH / 2);
            });

            // Draw Exit Marker (Prominent)
            const scaledExit = scaleCoords(exitPoint.x, exitPoint.y);
            ctx.fillStyle = '#E63946';
            ctx.shadowColor = '#E63946';
            ctx.shadowBlur = 15;
            ctx.beginPath();
            ctx.arc(scaledExit.x, scaledExit.y, 20, 0, Math.PI * 2);
            ctx.fill();
            ctx.shadowBlur = 0;
            ctx.fillStyle = '#FFFFFF';
            ctx.font = 'bold 16px Inter';
            ctx.textAlign = 'center';
            ctx.fillText('EXIT', scaledExit.x, scaledExit.y + 5);

            // Start TTS guidance with the first instruction
            if (instructions.length > 0) {
                speak(instructions[0]);
            }
        }

        /**
         * Draws an arrow head on a canvas line segment.
         * @param {{x: number, y: number}} p1 - Start point.
         * @param {{x: number, y: number}} p2 - End point.
         * @param {number} size - Size of the arrowhead.
         * @param {string} color - Color of the arrowhead.
         */
        function drawArrow(p1, p2, size, color) {
            const dx = p2.x - p1.x;
            const dy = p2.y - p1.y;
            const angle = Math.atan2(dy, dx);
            const dist = Math.sqrt(dx * dx + dy * dy);

            // Draw arrow head only if segment is long enough
            if (dist > size * 2) {
                const centerP = {
                    x: p1.x + dx * 0.5,
                    y: p1.y + dy * 0.5
                };

                ctx.save();
                ctx.translate(centerP.x, centerP.y);
                ctx.rotate(angle);
                ctx.fillStyle = color;
                ctx.beginPath();
                ctx.moveTo(-size, -size / 2);
                ctx.lineTo(0, 0);
                ctx.lineTo(-size, size / 2);
                ctx.closePath();
                ctx.fill();
                ctx.restore();
            }
        }

        /**
         * Simulates a hazard event and triggers route recalculation.
         * @param {string} hazardType - 'fire', 'flood', or 'clear'.
         */
        function simulateHazard(hazardType) {
            currentHazard = hazardType;
            let message = '';

            switch (hazardType) {
                case 'fire':
                    message = 'WARNING: Fire detected near Main Exit! Route is being recalculated.';
                    break;
                case 'flood':
                    message = 'ALERT: Flood detected on the long path. Route diverted.';
                    break;
                case 'clear':
                    message = 'Hazards Cleared. Proceeding with the original safe route.';
                    break;
            }

            showNotifier(message, hazardType === 'clear' ? 'orange' : 'red');
            recalculateRoute(hazardType);
        }

        /**
         * Recalculates the route based on hazard.
         */
        function recalculateRoute(hazardType) {
            drawMap();
        }

        // Global animation loop for visual effects (like flashing hazards)
        function animate() {
            requestAnimationFrame(animate);
            if (document.getElementById('screen-map').classList.contains('hidden') === false) {
                // Redraw map periodically to simulate flashing hazards
                drawMap();
            }
        }


        // Initialize the app on the intro screen and ensure canvas is ready
        window.onload = () => {
            // Load voices, then start the animation loop
            if (speech.onvoiceschanged !== undefined) {
                speech.onvoiceschanged = setVoice;
            } else {
                setVoice();
            }

            navigateTo('intro');
            animate(); // Start the animation loop for flashing hazards and general updates
        };

    </script>
